41:         alias_method :path_and_extension_without_localization, :path_and_extension
42:         
43:         # Override because the original version is too minimalist
44:         def path_and_extension(template_path) #:nodoc:
45:           template_path_without_extension = template_path.sub(@@native_extensions, '')
46:           [ template_path_without_extension, $1 ]
47:         end
48:         
49:         def locate_localized_path(template_path, use_full_path)
50:           current_language = Language.current_language
51:           
52:           cache_key = "#{current_language}:#{template_path}"
53:           cached = @@localized_path_cache[cache_key]
54:           return cached if cached
55:           
56:           if use_full_path
57:             template_path_without_extension, template_extension = path_and_extension(template_path)
58:             
59:             if template_extension
60:               template_file_name = full_template_path(template_path_without_extension, template_extension)
61:             else
62:               template_extension = pick_template_extension(template_path).to_s
63:               template_file_name = full_template_path(template_path, template_extension)
64:             end
65:           else
66:             template_file_name = template_path
67:             template_extension = template_path.split('.').last
68:           end
69:           
70:           pn = Pathname.new(template_file_name)
71:           dir, filename = pn.dirname, pn.basename('.' + template_extension)
72:           
73:           localized_path = dir + "#{filename}.#{current_language}.#{template_extension}"
74:           
75:           unless localized_path.exist?
76:             localized_path = template_file_name
77:           end
78:           
79:           @@localized_path_cache[cache_key] = localized_path.to_s
80:         end
